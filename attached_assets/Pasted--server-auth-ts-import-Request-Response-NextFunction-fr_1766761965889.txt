// server/auth.ts
import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';
import bcrypt from 'bcrypt';

// Get from environment or use default (CHANGE IN PRODUCTION!)
const JWT_SECRET = process.env.JWT_SECRET || 'CHANGE_THIS_IMMEDIATELY_IN_PRODUCTION_MIN_32_CHARS';
const JWT_EXPIRY = '7d'; // Token expires in 7 days

// User type for JWT payload
export interface AuthUser {
  id: string;
  username: string;
  email: string;
  role: 'admin' | 'user';
}

// Extend Express Request to include user
export interface AuthRequest extends Request {
  user?: AuthUser;
}

/**
 * Generate JWT token for user
 */
export function generateToken(user: AuthUser): string {
  return jwt.sign(
    { 
      id: user.id, 
      username: user.username, 
      email: user.email, 
      role: user.role 
    },
    JWT_SECRET,
    { expiresIn: JWT_EXPIRY }
  );
}

/**
 * Verify JWT token and return user data
 * ðŸ†• Now with better error logging
 */
export function verifyToken(token: string): AuthUser | null {
  try {
    const decoded = jwt.verify(token, JWT_SECRET) as AuthUser;
    return decoded;
  } catch (error) {
    // ðŸ†• Better error logging
    if (error instanceof jwt.TokenExpiredError) {
      console.log('Token expired');
    } else if (error instanceof jwt.JsonWebTokenError) {
      console.log('Invalid token');
    } else {
      console.log('Token verification error:', error);
    }
    return null;
  }
}

/**
 * ðŸ†• Refresh an expired token
 * Useful for keeping users logged in
 */
export function refreshToken(oldToken: string): string | null {
  try {
    const decoded = jwt.verify(oldToken, JWT_SECRET, { ignoreExpiration: true }) as AuthUser;
    // Issue new token with same user data
    return generateToken(decoded);
  } catch (error) {
    console.log('Token refresh failed:', error);
    return null;
  }
}

/**
 * Hash password using bcrypt
 */
export async function hashPassword(password: string): Promise<string> {
  const saltRounds = 10;
  return await bcrypt.hash(password, saltRounds);
}

/**
 * Compare plaintext password with hash
 */
export async function comparePassword(
  password: string, 
  hash: string
): Promise<boolean> {
  return await bcrypt.compare(password, hash);
}

/**
 * Middleware: Require authentication
 * Use on routes that need logged-in users
 */
export function requireAuth(
  req: AuthRequest, 
  res: Response, 
  next: NextFunction
): void {
  const authHeader = req.headers.authorization;
  
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    res.status(401).json({ error: 'Authentication required' });
    return;
  }
  
  const token = authHeader.substring(7); // Remove 'Bearer '
  const user = verifyToken(token);
  
  if (!user) {
    res.status(401).json({ error: 'Invalid or expired token' });
    return;
  }
  
  // Attach user to request
  req.user = user;
  next();
}

/**
 * Middleware: Require admin role
 * Use on routes that need admin access
 */
export function requireAdmin(
  req: AuthRequest, 
  res: Response, 
  next: NextFunction
): void {
  if (!req.user) {
    res.status(401).json({ error: 'Authentication required' });
    return;
  }
  
  if (req.user.role !== 'admin') {
    res.status(403).json({ error: 'Admin access required' });
    return;
  }
  
  next();
}

/**
 * Optional: Middleware to attach user if token exists
 * Use on routes that work with or without auth
 */
export function optionalAuth(
  req: AuthRequest, 
  res: Response, 
  next: NextFunction
): void {
  const authHeader = req.headers.authorization;
  
  if (authHeader && authHeader.startsWith('Bearer ')) {
    const token = authHeader.substring(7);
    const user = verifyToken(token);
    if (user) {
      req.user = user;
    }
  }
  
  next();
}

/**
 * ðŸ†• Middleware: Try to refresh token if expired
 * Automatically refreshes expired tokens
 */
export function autoRefreshAuth(
  req: AuthRequest, 
  res: Response, 
  next: NextFunction
): void {
  const authHeader = req.headers.authorization;
  
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    res.status(401).json({ error: 'Authentication required' });
    return;
  }
  
  const token = authHeader.substring(7);
  let user = verifyToken(token);
  
  // If token is invalid, try to refresh it
  if (!user) {
    const newToken = refreshToken(token);
    if (newToken) {
      user = verifyToken(newToken);
      if (user) {
        req.user = user;
        // Send new token in response header
        res.setHeader('X-New-Token', newToken);
        next();
        return;
      }
    }
    res.status(401).json({ error: 'Invalid or expired token' });
    return;
  }
  
  req.user = user;
  next();
}
